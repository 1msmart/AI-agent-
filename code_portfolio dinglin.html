API Setup — GPT Model
Wrapper to query the UF AI gateway; configurable model and temperature

library(httr); library(jsonlite)
library(dplyr); library(tidyverse); library(purrr); library(readxl)

api_key  <- "YOUR_API_KEY"
base_url <- "https://api.ai.it.ufl.edu/v1/chat/completions"

ask <- function(user.content, sys.content,
                 m = "gpt-oss-120b", tem = 0.5) {

  r <- POST(base_url,
    add_headers(Authorization = paste("Bearer", api_key),
                "Content-Type" = "application/json"),
    body = list(
      model    = m,
      messages = list(
        list(role = "system", content = sys.content),
        list(role = "user",   content = user.content)
      )
    ),
    encode      = "json",
    temperature = tem,   # 0 = deterministic, 1 = creative
    max_tokens  = 500
  )

  if (status_code(r) == 200) {
    res    <- fromJSON(content(r, "text"))
    answer <- res$choices$message$content
    writeLines(answer)
    return(answer)
  } else {
    cat("Error:", content(r, "text"))
  }
}

CE Choice Function — Prompt Engineering
Persona-aware system prompt + attribute-level user prompt; model must respond with only 1, 2, or 3
Prompt Design R
R · Conjoint Prompt 
CE_Choice <- function(gender, age, edu, income, price_paid,
                       price1, origins1, method1, review1, bestuse1,
                       price2, origins2, method2, review2, bestuse2) {
  ask(
    sys.content = "You are a consumer making a real purchase decision while
shopping for fresh strawberries in a grocery store. You must select one of
three options: Option 1, Option 2, or Option 3 (neither).

Strawberry price typically ranges from $0.99-$8.99/lb. Consider budget
constraints -- higher price means less income for other products.

You are the SAME respondent making 14 shopping decisions in sequence.
Maintain consistency with your demographics and preferences throughout.

ATTRIBUTE DEFINITIONS:
- Organic: organic farming, limited synthetic inputs
- Locally produced: grown in the same state as marketed
- Naturally grown: non-profit alternate farm assurance certification
- Customer Review: avg. sensory rating (★★★ = best)",

    user.content = paste(
      "You are", gender, ",", age, "years old, with a", edu,
      "degree, and annual income of $", income, ".",
      "Previously paid $", price_paid, "for 1 lb of fresh strawberries.",
      "Respond with ONLY the number 1, 2, or 3.",

      "Option 1:", method1, ", from", origins1,
      ",", bestuse1, ",", review1, "review. Price: $", price1, "/lb.",

      "Option 2:", method2, ", from", origins2,
      ",", bestuse2, ",", review2, "review. Price: $", price2, "/lb.",

      "Option 3: I would not choose either product.",
      sep = " "
    )
  )
}

Baseline AI Predictions — Resumable Pipeline
Iterates over all respondent × choice-set combos with checkpoint saving so long API runs can resume after interruption
R LLM Inference Automation
R · Resumable Loop 
CE_data_for_AI_resumable <- function(ce_data,
                                       save_interval = 100,
                                       output_file = "ai_results_temp.rds") {

  # Pivot to wide: one row per respondent x choice set
  choice_sets <- ce_data %>%
    filter(Alt %in% c(1, 2)) %>%
    group_by(RID, Set) %>%
    summarise(
      gender       = first(Gender), age          = first(Age),
      edu          = first(Education), income     = first(Income),
      price_paid   = first(Price_paid), human_choice = first(Choose),
      price1       = Price[Alt == 1], origins1 = Origin[Alt == 1],
      method1      = Method[Alt == 1], review1  = Review[Alt == 1],
      bestuse1     = Dates[Alt == 1],
      price2       = Price[Alt == 2], origins2 = Origin[Alt == 2],
      method2      = Method[Alt == 2], review2  = Review[Alt == 2],
      bestuse2     = Dates[Alt == 2],
      .groups = "drop"
    )

  # Resume from checkpoint if it exists
  if (file.exists(output_file)) {
    cat("Checkpoint found. Resuming...\n")
    choice_sets <- readRDS(output_file)
    start_row   <- sum(!is.na(choice_sets$AI_Choice)) + 1
  } else {
    choice_sets$AI_Choice <- NA_character_
    start_row <- 1
  }

  total_rows <- nrow(choice_sets)

  for (i in start_row:total_rows) {
    row <- choice_sets[i, ]
    choice_sets$AI_Choice[i] <- CE_Choice(
      gender = row$gender, age = row$age, edu = row$edu, income = row$income,
      price_paid = row$price_paid,
      price1 = row$price1, origins1 = row$origins1, method1 = row$method1,
      review1 = row$review1, bestuse1 = row$bestuse1,
      price2 = row$price2, origins2 = row$origins2, method2 = row$method2,
      review2 = row$review2, bestuse2 = row$bestuse2
    )
    if (i %% save_interval == 0 || i == total_rows) {
      saveRDS(choice_sets, output_file)
      cat(sprintf("Checkpoint: %d / %d complete\n", i, total_rows))
    }
  }
  return(choice_sets)
}

# -- Split by respondent ID ----------------------------------------
CE_data_train_test_split <- function(ce_data, train_proportion = 0.7) {
  set.seed(123)
  all_rids   <- unique(ce_data$RID)
  train_rids <- sample(all_rids, floor(length(all_rids) * train_proportion))
  test_rids  <- setdiff(all_rids, train_rids)

  list(
    train_data        = ce_data %>% filter(RID %in% train_rids),
    test_data         = ce_data %>% filter(RID %in% test_rids),
    train_respondents = train_rids, test_respondents = test_rids
  )
}

# -- Summarise training-set preference patterns -------------------
summarize_training_patterns <- function(train_data) {
  wide <- train_data %>%
    filter(Alt %in% c(1, 2)) %>%
    group_by(RID, Set) %>%
    summarise(across(everything(), first), .groups = "drop")

  list(
    overall      = wide %>% count(choice) %>%
                   mutate(pct = n / sum(n) * 100),

    price_effect = wide %>% filter(choice %in% c(1, 2)) %>%
                   mutate(chosen_price = ifelse(choice == 1, price1, price2)) %>%
                   summarise(avg = mean(chosen_price), sd  = sd(chosen_price),
                             mn  = min(chosen_price),  mx  = max(chosen_price)),

    method_pref  = wide %>% filter(choice %in% c(1, 2)) %>%
                   mutate(chosen = ifelse(choice == 1, method1, method2)) %>%
                   count(chosen) %>% mutate(pct = n / sum(n) * 100) %>%
                   arrange(desc(pct))
  )
}

Within-Person Training — First 7 → Last 7
Each respondent's first 7 actual choices (with full attribute detail) are injected as history; predict their remaining 7 choice sets
In-Context Learning R Within-Person
R · Within-Person Pipeline 
CE_data_within_person <- function(ce_data, n_training_sets = 7,
                                   save_interval = 100,
                                   output_file = "ai_results_within_person.rds") {

  all_sets <- ce_data %>%
    filter(Alt %in% c(1, 2)) %>%
    group_by(RID, Set) %>%
    summarise(
      gender = first(Gender), age    = first(Age),
      income = first(Income),  edu    = first(Education),
      human_choice = first(Choose),
      price1  = Price[Alt == 1], method1  = Method[Alt == 1],
      origins1= Origin[Alt == 1], review1  = Review[Alt == 1],
      bestuse1= Dates[Alt == 1],
      price2  = Price[Alt == 2], method2  = Method[Alt == 2],
      origins2= Origin[Alt == 2], review2  = Review[Alt == 2],
      bestuse2= Dates[Alt == 2],
      .groups = "drop"
    ) %>% arrange(RID, Set)

  train_sets <- all_sets %>% filter(Set <= n_training_sets)
  test_sets  <- all_sets %>% filter(Set >  n_training_sets)
  test_sets$AI_Choice <- NA_character_

  for (i in 1:nrow(test_sets)) {
    row     <- test_sets[i, ]
    history <- train_sets %>% filter(RID == row$RID) %>% arrange(Set)

    # Build natural-language history string
    lines <- sapply(seq_len(nrow(history)), function(j) {
      h <- history[j, ]
      if      (h$human_choice == 1)
        sprintf("Set %d: Option 1 — %s, %s, %s, %s review, $%.2f/lb",
                h$Set, h$method1, h$origins1, h$bestuse1, h$review1, h$price1)
      else if (h$human_choice == 2)
        sprintf("Set %d: Option 2 — %s, %s, %s, %s review, $%.2f/lb",
                h$Set, h$method2, h$origins2, h$bestuse2, h$review2, h$price2)
      else
        sprintf("Set %d: Option 3 (neither)", h$Set)
    })

    history_text <- paste0(
      "\nYour previous ", nrow(history), " choices:\n",
      paste(lines, collapse = "\n"),
      "\n\nNow making choice set ", row$Set, " of 14.",
      "\nIdentify your preference patterns and stay consistent."
    )

    test_sets$AI_Choice[i] <- CE_Choice_WithHistory(
      gender = row$gender, age = row$age, edu = row$edu,
      income = row$income, price_paid = row$price_paid,
      price1 = row$price1, origins1 = row$origins1, method1 = row$method1,
      review1 = row$review1, bestuse1 = row$bestuse1,
      price2 = row$price2, origins2 = row$origins2, method2 = row$method2,
      review2 = row$review2, bestuse2 = row$bestuse2,
      history_text = history_text
    )
  }

  test_sets$AI_Choice_num <- as.numeric(test_sets$AI_Choice)
  accuracy <- mean(test_sets$AI_Choice_num == test_sets$human_choice, na.rm = TRUE)
  cat(sprintf("Within-Person Accuracy: %.2f%%\n", accuracy * 100))
  print(table(Actual = test_sets$human_choice, Predicted = test_sets$AI_Choice_num))
  return(list(results = test_sets, accuracy = accuracy))
}
